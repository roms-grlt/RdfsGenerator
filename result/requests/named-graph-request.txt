PREFIX unified: <http://example.org/unified/>
PREFIX amazon: <http://example.org/amazon/>
PREFIX netflix: <http://example.org/netflix/>

SELECT ?graph ?film ?title ?rating
WHERE {
  GRAPH ?graph {
    ?film a unified:film ;
          unified:title ?title ;
          unified:rating ?rating .
  }
  FILTER(?rating > 4.0)
}
ORDER BY DESC(?rating)
LIMIT 15
